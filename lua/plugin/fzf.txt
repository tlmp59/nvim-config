return {
	{
		"ibhagwan/fzf-lua",
		event = "VimEnter",
		config = function()
			-- calling `setup` is optional for customization
			require("fzf-lua").setup({
				defaults = {
					git_icons = false,
					file_icons = false,
					color_icons = false,
				},
			})

			local fzf = require("fzf-lua")
			local M = vim.keymap.set
			M("n", "<leader>fzf", fzf.files)
			M("n", "<leader>fzb", fzf.buffers)
			M("n", "<leader>fzg", fzf.live_grep)
		end,
	},
}
